\section{Scope, engines, and discipline}\label{sec:scope}

All statements below concern the objects of \cite{relcharge}: for a monic
$p\in\Z[x]$ with $p(0)\neq 0$ and roots $\alpha_1,\dots,\alpha_n\in\mathbb{C}$,
the \emph{angle} $t(\alpha)\in\R/\Z$ is the normalized argument, two roots are
\emph{coherent}, $\alpha\approx\beta$, iff
$t_{\mathrm{rel}}(\alpha,\beta)=t(\alpha)-t(\beta)\in\Q/\Z$, a \emph{shell} is a
level set of the modulus, and the \emph{relational type} is the partition of the
root set into coherence classes together with the class offsets. The
\emph{ratio object} is
\[
\Rat_p(x)\;=\;\operatorname{prim}\,\Res_y\bigl(p(y),\,p(xy)\bigr)\in\Z[x].
\]

Two engines were used, matching the ledger-M standard of \cite{relcharge}:
\sy~1.14.0 on Python~3.12.3 (arbitrary-precision Python integers), and
\gp~2.15.4 on GMP~6.3.0 (amd64, gcc~13.2.0 build). \emph{No floating-point value
crosses any decision boundary}: every equality, divisibility, and inequality
below is settled by exact division in $\Z[x]$, Sturm counts at rational
endpoints, or rational-interval certificates. Where residues are pruned modulo
the primes $10^9{+}7$ and $998244353$, pruning discards \emph{negatives only};
every positive is confirmed by exact division over $\Z$. Tags follow house
convention: \F\ (machine-verified theorem / exact per-instance certificate),
\C\ (exact computation over a stated finite range), \Pl, \Dc, \Op.

\section{The scan as a decision procedure}\label{sec:scan}

\begin{proposition}[ratio object]\label{prop:rat}
Let $p$ be monic of degree $n$ with $p(0)\neq0$ and roots $\alpha_1,\dots,\alpha_n$
(with multiplicity). Then
\[
\Res_y\bigl(p(y),p(xy)\bigr)
=\bigl((-1)^n p(0)\bigr)^{n}\prod_{i,j}\Bigl(x-\frac{\alpha_j}{\alpha_i}\Bigr),
\]
so $\Rat_p$ has degree exactly $n^2$, root multiset the $n^2$ ordered ratios,
and integer coefficients. Moreover the multiplicity of $(x-1)$ in $\Rat_p$
equals $\sum_{\text{distinct roots}}m_\alpha^{2}$; in particular it equals $n$
iff $p$ is squarefree. \F
\end{proposition}

\begin{proof}
With $f=p(y)$ monic and $g=p(xy)$ of $y$-degree $n$,
$\Res_y(f,g)=\lc(f)^{\deg g}\prod_{p(\alpha_i)=0}p(x\alpha_i)$ and
$p(x\alpha_i)=\prod_j(x\alpha_i-\alpha_j)=\alpha_i^{\,n}\prod_j(x-\alpha_j/\alpha_i)$;
collect $\prod_i\alpha_i^{\,n}=((-1)^np(0))^n\neq0$. The diagonal count is
$\#\{(i,j):\alpha_i=\alpha_j\}=\sum m_\alpha^2$.
\end{proof}

\begin{lemma}[contact criterion]\label{lem:contact}
Suppose $|\alpha|=|\beta|$. Then $\alpha\approx\beta$ iff $\alpha/\beta$ is a
root of unity; and if $\alpha/\beta$ is a primitive $M$-th root of unity then
$\Phi_M\mid\Rat_p$ in $\Z[x]$ (Gauss). Conversely $\Phi_M\mid\Rat_p$ exhibits a
primitive $M$-th root of unity among the ratios. \F
\end{lemma}

\begin{lemma}[completeness bound]\label{lem:totient}
For every $M\ge1$, $2\varphi(M)^2\ge M$, with equality iff $M=2$. Hence
$\Phi_M\mid P$ forces $M\le 2(\deg P)^2$; the complete contact scan of $\Rat_p$
is a finite decision procedure with bound $2n^4$, entirely in $\Z[x]$. \F
\end{lemma}

\begin{proof}
Odd prime power: $\varphi(p^b)=p^{b-1}(p-1)\ge p^{b-1}\sqrt{p}=p^{\,b-\frac12}\ge p^{b/2}$,
using $p-1\ge\sqrt p\iff p^2-3p+1\ge0$ for $p\ge3$. Two-part, $a\ge1$:
$\varphi(2^a)=2^{a-1}\ge 2^{(a-1)/2}$, equality iff $a=1$. Multiply:
$\varphi(M)\ge\sqrt{M/2}$, tight only at $M=2$. The equality case was swept
exactly to $M\le 2\cdot10^{5}$: no violations, unique tight case $M=2$. \C
\end{proof}

\begin{remark}[$\nu$-criterion, two-part caveat]\label{rem:nu2part}
For a cross-shell pair set $\mu=\alpha/\beta$ and $\nu=\mu/\bar\mu$; then
$t(\nu)=2\,t_{\mathrm{rel}}$, so \emph{rationality} of the offset is decided
exactly while its \emph{order} is read only up to the $2$-part
(\cite[Rem.~7.6]{relcharge}).
\end{remark}

Engine corroboration of \S\ref{sec:scan} on the canonical ledger (signatures
$A$--$X$ of \cite{relcharge}) reproduced every row, including the Lehmer scan
($\deg\Rat=100$, bound $20000$, $\{\Phi_1^{10}\}$, $0.89$\,s against the
deposit's $1.1$\,s), the mixed scan $\Rat_{L,\beta_4}$ ($\deg 40$, empty), the
Salem certifications of $\beta_4,S_6,S_8,L$ by trace-polynomial Sturm patterns
$(1,0,n/2-1)$, the two-route identity

$\operatorname{charpoly}(C_{x^4-x+1}\otimes C_{x^4-x+1})=S_6^{\,2}\cdot(x^4+2x^2-x+1)$

\section{Modulus pinning and its sharpness}\label{sec:pinning}

\begin{theorem}[modulus pinning; {\cite[Thm.~7.13]{relcharge}}, proof included for
self-containment]\label{thm:pin}
Let $p\in\Z[x]$ be irreducible with a root $\alpha_0$ whose modulus is attained
by no other root. Then no two distinct roots of $p$ differ by a root-of-unity
factor. Mixed form: if $q\neq p$ is irreducible and no root of $q$ has modulus
$|\alpha_0|$, then no torsion ratio exists between a root of $p$ and a root of
$q$. \F
\end{theorem}

\begin{proof}
Suppose $\alpha\neq\beta$, $p(\alpha)=p(\beta)=0$, $\alpha=\zeta\beta$ with
$\zeta^M=1$. Transitivity of $G=\Gal$ of the splitting field (the \emph{only}
use of irreducibility) supplies $\sigma$ with $\sigma\alpha=\alpha_0$. Since
$(\sigma\zeta)^M=1$, $\sigma\zeta$ is torsion, hence $|\sigma\zeta|=1$; thus
$|\sigma\beta|=|\alpha_0|$, forcing $\sigma\beta=\alpha_0=\sigma\alpha$ and
$\alpha=\beta$. For the mixed form, work in the splitting field $L$ of $pq$;
restriction $\Gal(L/\Q)\twoheadrightarrow\Gal(K_p/\Q)$ lifts transitivity, and
$\sigma\beta=\sigma\alpha/\sigma\zeta$ is a $q$-root at the excluded modulus.
\end{proof}

\begin{remark}[proof audit]\label{rem:audit}
(i)~The complex modulus is \emph{not} Galois-equivariant and the proof never
needs it to be; only the Galois-stable predicate ``is torsion'' plus $|{\cdot}|=1$
for torsion enters. (ii)~The pinned root need not participate in the offending
pair. (iii)~``Uniquely attained'', not ``maximal'', is the hypothesis; a Salem
polynomial carries two pins ($\tau$ and $1/\tau$). Corollaries
\cite[7.14, 7.15]{relcharge} (Salem inertness; no cross-locking of distinct
Salems) follow as in the parent paper and were re-derived and engine-checked.
\end{remark}

\begin{proposition}[hypothesis-necessity witness $Z^{*}$]\label{prop:zstar}
Let $f_1=x^2-x-1$, $f_2=x^2+x-1$, and
$Z^{*}=f_1f_2=x^4-3x^2+1$. Each $f_i$ is irreducible and \emph{both} of its
root moduli ($\varphi$ and $\varphi^{-1}$) are uniquely attained within that
factor, so per-factor pinning holds; yet $Z^{*}$ carries the torsion ratio $-1$
between the roots $\varphi$ (of $f_1$) and $-\varphi$ (of $f_2$) at the shared
modulus $\varphi$. Consequently: (i)~irreducibility cannot be dropped from
Theorem~\ref{thm:pin}; (ii)~the disjoint-modulus clause cannot be dropped from
its mixed form. The complete scan confirms
$\Rat_{Z^{*}}\leadsto\{\Phi_1^4,\Phi_2^4\}$. \F
\end{proposition}

\begin{proof}
$f_1$ has roots $\varphi,-1/\varphi$; $f_2$ has roots $1/\varphi,-\varphi$.
Moduli per factor: $\{\varphi,\varphi^{-1}\}$, each once. Across factors,
$\varphi/(-\varphi)=-1$ is torsion at the shared modulus $\varphi$. The
same-polynomial form fails only through transitivity (absent for the reducible
product); the mixed form fails only through its shared-modulus clause. The
signature $\{\Phi_1^4,\Phi_2^4\}$ ($\Phi_1$ from the diagonal, $\Phi_2$ from the
two antipodal pairs, ordered) is engine-confirmed in both stacks.
\end{proof}

\begin{remark}
$Z^{*}$ complements the parent paper's sharpness example $x^4-2$ (which shows
the pinning \emph{hypothesis} necessary for an irreducible input): together the
two witnesses isolate every hypothesis of Theorem~\ref{thm:pin} and its mixed
form. Candidate placement: a remark beside \cite[Ex.~7.19]{relcharge}.
\end{remark}

\section{The real-pair $\nu$-reduction; inertness below two non-real shells}
\label{sec:reduction}

\begin{lemma}[real-pair $\nu$-reduction]\label{lem:redu}
Let $O$ be a conjugation-closed set of nonzero algebraic numbers, $\beta\in O$
real, $\alpha\in O$ non-real. Then
\[
\alpha\approx\beta \iff \alpha/\bar\alpha \text{ is a root of unity.}
\]
If $O$ is the root set of $p$, then $\alpha/\bar\alpha$ is a \emph{root of
$\Rat_p$}; hence every real$\times$non-real coherence question --- same shell or
cross shell --- is decided by the complete contact scan of $\Rat_p$ alone. \F
\end{lemma}

\begin{proof}
$t(\beta)\in\{0,\tfrac12\}$, so $t(\alpha)-t(\beta)\in\Q\iff t(\alpha)\in\Q
\iff 2t(\alpha)\in\Q$. The number $\alpha/\bar\alpha$ is unimodular with
$t(\alpha/\bar\alpha)=2t(\alpha)$; unimodular with rational angle is exactly
torsion (Lemma~\ref{lem:contact}). Since $\bar\alpha\in O$, the quantity is an
ordered ratio, i.e.\ a root of $\Rat_p$. \cite[Ex.~7.17(iii)]{relcharge} is the
$\beta_4$ instance of this statement.
\end{proof}

\begin{theorem}[Pisot inertness below two non-real shells]\label{thm:pisot1pair}
Every Pisot number whose minimal polynomial has at most one pair of non-real
conjugates --- in particular every Pisot number of degree $\le 4$, and every
quintic Pisot of pattern $(\theta,r_1,r_2,\lambda,\bar\lambda)$ --- is
relationally inert: its type is one rational block containing exactly the real
conjugates, plus singletons $\{\lambda\},\{\bar\lambda\}$ when the pair exists. \F
\end{theorem}

\begin{proof}
The dominant root $\theta>1$ pins its modulus (all other conjugates lie in the
open unit disk), so Theorem~\ref{thm:pin} excludes every torsion ratio between
distinct conjugates; in particular $\lambda/\bar\lambda$ is not torsion, so by
Lemma~\ref{lem:redu} no real$\times$non-real pair is coherent, and
$\lambda\not\approx\bar\lambda$ within the pair's shell. Real$\times$real pairs
have angle difference in $\tfrac12\Z\subset\Q$ and form the rational block.
With at most one non-real pair there is no non-real cross-shell pair left to
decide.
\end{proof}

\begin{corollary}\label{cor:residue-location}
The open cross-shell residue of \cite[Cor.~7.16]{relcharge} is empty through
the patterns covered by Theorem~\ref{thm:pisot1pair}. Its first live case is
the quintic two-pair pattern $(\theta,\lambda_1,\bar\lambda_1,\lambda_2,\bar\lambda_2)$. \F
\end{corollary}

\section{The residue at two non-real shells}\label{sec:residue}

Throughout this section $p$ is an irreducible Pisot quintic of two-pair pattern;
write $u_i=\lambda_i/\bar\lambda_i$ (unimodular, and by
Proposition~\ref{prop:rat} a root of $\Rat_p$), and
$\Rat_p^{\circ}=\Rat_p/(x-1)^5$.

\begin{proposition}[$\nu$ inventory; level-2 formulation]\label{prop:nuinv}
For distinct shells $|\lambda_1|\neq|\lambda_2|$:
\[
\lambda_1\approx\lambda_2\iff u_1\bar u_2\ \text{torsion},\qquad
\lambda_1\approx\bar\lambda_2\iff u_1u_2\ \text{torsion}.
\]
Hence a mirrored cross-shell class exists iff two unimodular roots of
$\Rat_p^{\circ}$ differ multiplicatively by a root of unity. The degenerate
case $u_1=u_2$ (offset in $\tfrac12\Z$) forces a repeated root of
$\Rat_p^{\circ}$ and is detected by its squarefreeness. \F
\end{proposition}

\begin{proof}
$t(u_1\bar u_2)=2(t_1-t_2)$ and $t(u_1u_2)=2(t_1+t_2)$; rationality transfers
both ways (Remark~\ref{rem:nu2part} affects only the order, not the decision).
Both $u_2$ and $u_2^{-1}=\bar u_2$ are roots of $\Rat_p$. If $u_1=u_2$ as
values, two distinct ordered ratios coincide.
\end{proof}

\begin{proposition}[no level-2 pin]\label{prop:nopin}
In the two-pair pattern with distinct shells, the modulus multiset of
$\Rat_p^{\circ}$ is
\[
\Bigl\{\tfrac{\theta}{|\lambda_1|},\tfrac{\theta}{|\lambda_2|},
\tfrac{|\lambda_1|}{\theta},\tfrac{|\lambda_2|}{\theta}\ (\times2\ \text{each}),\
\tfrac{|\lambda_1|}{|\lambda_2|},\tfrac{|\lambda_2|}{|\lambda_1|}\ (\times4\ \text{each}),\
1\ (\times4)\Bigr\},
\]
every value attained at least twice (coincidences only merge classes). Hence
Theorem~\ref{thm:pin} is inapplicable one level up: no transport argument closes
the residue, and a scan is genuinely required. \F
\end{proposition}

\begin{proposition}[the $S^{*}/C_2$ instrument]\label{prop:c2}
Let $S^{*}$ be the product of the self-reciprocal irreducible $\Z$-factors of
$\Rat_p^{\circ}$ that carry at least one unimodular root (certified per factor
by the Sturm count of the exact trace square on $(-2,2)$), $d=\deg S^{*}$, and
\[
C_2(x)\;=\;\Res_y\bigl(S^{*}(y),\,y^{\,d}S^{*}(x/y)\bigr).
\]
Then: (a)~the minimal polynomial of any unimodular algebraic number $u\neq\pm1$
is self-reciprocal (its Galois orbit is inversion-closed, since
$u^{-1}=\bar u$ lies in it), so $u_1,\bar u_1,u_2,\bar u_2$ are roots of
$S^{*}$; (b)~$C_2$ is monic of degree $d^{\,2}$ with root multiset all ordered
products of $S^{*}$-roots; (c)~\emph{negative certificate}: if the complete
contact scan of $C_2$ returns exactly $\{\Phi_1^{\,d}\}$, then no product of two
$S^{*}$-roots is a root of unity; in particular $u_1\bar u_2$ and $u_1u_2$ are
not torsion and no mirrored cross-shell class exists; (d)~the forced
$\Phi_1$-multiplicity is exactly $d$ (each root pairs uniquely with its inverse;
$\pm1$ are excluded from $S^{*}$ by level-1 pinning), so any excess or any
$\Phi_M$, $M\ge2$, is a signal. \F
\end{proposition}

\begin{proposition}[shell detector]\label{prop:shells}
For the two-pair pattern, the number of unimodular roots of $\Rat_p^{\circ}$
equals $4$ iff $|\lambda_1|\neq|\lambda_2|$ and $12$ iff
$|\lambda_1|=|\lambda_2|$; the same-shell case is already closed by level-1
pinning (all its cross pairs are ordered ratios of conjugates). \F
\end{proposition}

\F\ per instance; box coverage \C.

\section{Verification-harness findings}\label{sec:harness}

Both discovered by a pinned-tally failure (the truncated run produced the

\begin{itemize}
\item[\textbf{H1.}] \gp\ terminates top-level statements at newline: a function
body split across physical lines without braces silently truncates. Rule:
multi-line \gp\ definitions must be brace-wrapped; single-line otherwise.
\item[\textbf{H2.}] \texttt{gp} exits $0$ after load errors, emitting
\texttt{***} diagnostics to stderr only (``skipping file''). Rule: harnesses
must treat any non-benign \texttt{***} stderr line as fatal.
\item[\textbf{H3.}] Positive pattern: the dual-path
$\mathrm{sig}(P)=\mathrm{scan}_{\mathrm{td}}(P)\overset{!}{=}\mathrm{scan}_{\mathrm{fp}}(P)$
assertion (trial division vs.\ factorization) as the in-engine standard for
\gp-side scans.
\item[\textbf{H4.}] \Dc\ Container-class observation, disclosed rather than
promoted: one execution container reaped backgrounded processes at call
teardown nondeterministically (the N3 driver survived only because its launch
call outlived it; the N2 driver was killed mid-import with an empty log); a
second container of the same family did \emph{not} reproduce it (five
harness-tracked background runs, up to $15$ minutes, all completed). Binding
pattern regardless: foreground execution with checkpointed, restart-safe
drivers.
\end{itemize}

\begin{remark}[shell-detector semantics, corrected]\label{rem:detector}
Proposition~\ref{prop:shells} counts unimodular roots \emph{with
multiplicity} ($4$ vs.\ $12$). An implementation reading the trace-square
Sturm count on $(-2,2)$ counts \emph{distinct} values instead; in the merged
case the $12$ ratios provably collapse to at most $4$ distinct trace values
(the cross ratios pair off), so a merged branch keyed to the distinct count
$6$ is unreachable, and a torsion-merged configuration can present the
distinct-shell count silently. The merged case is nonetheless excluded on
every instance by the joint certificate actually asserted --- distinct-shell
count, $\Phi_2$-free, no higher cyclotomic content, no count anomaly ---
since any silent collapse forces torsion ratios that the $C_2$/torsion scan
exhibits. In the $[-2,2]^5$ run the verified squarefreeness of
$\Rat_p^{\circ}$ already equates the two counts. \F
\end{remark}

\subsection{Predictions}
\begin{conjecture}[Pisot inertness, general]\label{conj:pisot}
Every Pisot number is relationally inert: its type is the rational block of its
real conjugates plus non-real singletons. Equivalently, for conjugate-ratio
values $u_i=\lambda_i/\bar\lambda_i$ on distinct shells, $u_iu_j^{\pm1}$ is
never a root of unity. \Op
\end{conjecture}

Status: \F\ for at most one non-real pair (Theorem~\ref{thm:pisot1pair});
\C-empty on the first $67$ live instances (Theorem~\ref{thm:box}) and on all
$313$ two-pair instances of $[-3,3]^5$ (\S\ref{sec:boxext});
Proposition~\ref{prop:nopin} shows transport cannot prove it, so a genuinely
new mechanism (heights, multiplicative-relation bounds) is required.
